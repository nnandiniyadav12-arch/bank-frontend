const fmt = n => new Intl.NumberFormat('en-IN',{style:'currency',currency:'INR',maximumFractionDigits:0}).format(n);
const fmtNum = n => new Intl.NumberFormat('en-IN').format(n);
const id4 = (prefix,n) => `${prefix}-${String(n).padStart(4,'0')}`;
const getCust = id => DB.customers.find(c=>c.id===id)||{name:'Unknown',id:0};
const getBranch = id => DB.branches.find(b=>b.id===id)||{name:'Unknown'};
const getBanker = id => DB.bankers.find(b=>b.id===id)||{name:'Unknown'};
const getAcct = id => DB.accounts.find(a=>a.id===id)||{balance:0,type:'',customer_id:0};
const loanPct = l => Math.round(((l.amount-l.remaining)/l.amount)*100);
const isOverdue = l => new Date(l.due)<new Date();
const custAccounts = cid => DB.accounts.filter(a=>a.customer_id===cid);
const custTotalBalance = cid => custAccounts(cid).reduce((s,a)=>s+a.balance,0);
const initials = name => name.split(' ').map(w=>w[0]).join('').slice(0,2).toUpperCase();

function toast(msg,type='accent'){
  const w=document.getElementById('toast-wrap');
  const t=document.createElement('div');
  t.className='toast';
  const colors={'accent':'var(--accent)','red':'var(--red)','blue':'var(--blue)','amber':'var(--amber)'};
  t.innerHTML=`<span class="toast-dot" style="background:${colors[type]||colors.accent}"></span>${msg}`;
  w.appendChild(t);
  setTimeout(()=>{t.style.opacity='0';t.style.transform='translateX(20px)';setTimeout(()=>t.remove(),300)},3000);
}

function switchView(name,el){
  document.querySelectorAll('.view').forEach(v=>v.classList.remove('active'));
  document.querySelectorAll('.nav-item').forEach(n=>n.classList.remove('active'));
  document.getElementById('view-'+name).classList.add('active');
  el.classList.add('active');
  const m=VIEW_META[name];
  document.getElementById('topbar-title').textContent=m.title;
  document.getElementById('topbar-sub').textContent=m.sub;
  renderView(name);
}

function renderView(name){
  if(name==='dashboard')renderDashboard();
  else if(name==='customers')renderCustomers();
  else if(name==='accounts')renderAccounts();
  else if(name==='transactions')renderTransactions();
  else if(name==='loans')renderLoans();
  else if(name==='creditcards')renderCreditCards();
  else if(name==='bankers')renderBankers();
  else if(name==='branches')renderBranches();
}

function recordPayment(){
  const amt=parseFloat(document.getElementById('pay-amount').value)||0;
  const method=document.getElementById('pay-method').value;
  const date=document.getElementById('pay-date').value;
  if(!amt||amt<=0){toast('Enter a valid amount','red');return}
  const loan=DB.loans.find(l=>l.id===activeLoanId);
  if(amt>loan.remaining){toast('Amount exceeds remaining balance','red');return}
  loan.remaining-=amt;
  const newId=Math.max(...DB.loanPayments.map(p=>p.id))+1;
  DB.loanPayments.push({id:newId,loan_id:activeLoanId,amount:amt,date,method});
  toast(`Payment of ${fmt(amt)} recorded!`,'accent');
  openLoanPay(activeLoanId);
  renderLoans();
}

function openModal(name){
  if(name==='customer'){
    const br=document.getElementById('nc-branch');
    br.innerHTML=DB.branches.map(b=>`<option value="${b.id}">${b.name}</option>`).join('');
  }
  if(name==='transaction'){

  }
  if(name==='loan'){
    document.getElementById('nl-customer').innerHTML=DB.customers.map(c=>`<option value="${c.id}">${c.name}</option>`).join('');
    document.getElementById('nl-banker').innerHTML=DB.bankers.map(b=>`<option value="${b.id}">${b.name} (${getBranch(b.branch_id).name})</option>`).join('');

  }
  if(name==='banker'){
    document.getElementById('nb-branch').innerHTML=DB.branches.map(b=>`<option value="${b.id}">${b.name}</option>`).join('');
  }
  document.getElementById('overlay-'+name).classList.add('open');
}

function saveCustomer(){
  const name=document.getElementById('nc-name').value.trim();
  const dob=document.getElementById('nc-dob').value;
  const mobile=document.getElementById('nc-mobile').value.trim();
  const branchId=parseInt(document.getElementById('nc-branch').value);
  const accType=document.getElementById('nc-acctype').value;
  const balance=parseFloat(document.getElementById('nc-balance').value)||0;
  if(!name||!dob||!mobile){toast('Please fill all required fields','red');return}
  if(!/^\d{10}$/.test(mobile)){toast('Enter a valid 10-digit mobile number','red');return}
  const newCid=Math.max(...DB.customers.map(c=>c.id))+1;
  DB.customers.push({id:newCid,name,dob,mobile,branch_id:branchId});
  const newAid=Math.max(...DB.accounts.map(a=>a.id))+1;
  DB.accounts.push({id:newAid,balance,type:accType,customer_id:newCid,branch_id:branchId,opened:new Date().toISOString().slice(0,10),status:'Active'});
  closeModal('customer');
  ['nc-name','nc-dob','nc-mobile','nc-balance'].forEach(id=>document.getElementById(id).value='');
  toast(`Customer ${name} registered with ${accType} account`);
  renderCustomers();renderDashboard();
}

function saveTransaction(){
  const accId=parseInt(document.getElementById('nt-account').value);
  const type=document.getElementById('nt-type').value;
  const amount=parseFloat(document.getElementById('nt-amount').value)||0;
  const desc=document.getElementById('nt-desc').value.trim()||'Manual Entry';
  const date=document.getElementById('nt-date').value||new Date().toLocaleString();
  if(!amount||amount<=0){toast('Enter a valid amount','red');return}
  const acc=DB.accounts.find(a=>a.id===accId);
  if(type==='Debit'&&acc.balance<amount){toast('Insufficient balance!','red');return}
  if(type==='Credit')acc.balance+=amount; else acc.balance-=amount;
  const newId=Math.max(...DB.transactions.map(t=>t.id))+1;
  DB.transactions.push({id:newId,account_id:accId,amount,type,date:date.replace('T',' '),desc});
  closeModal('transaction');
  document.getElementById('nt-amount').value='';document.getElementById('nt-desc').value='';
  toast(`${type} of ${fmt(amount)} posted successfully`,'accent');
  renderDashboard();
  if(document.getElementById('view-transactions').classList.contains('active'))renderTransactions();
  if(document.getElementById('view-accounts').classList.contains('active'))renderAccounts();
}
